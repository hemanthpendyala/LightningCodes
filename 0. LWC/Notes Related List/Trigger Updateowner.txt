Trigger Updateowner on case(after inset, after update){
set<id> OwnerId = new set<id> ();
set<id> caseid= new set<id> ();
List<case>  casesToUpdate = new list <case> ();
for (case caseobj: trigger.new ){
ownerid.add(caseobj.OwnerId)
caseid.add(caseobj.AccountId);
}
for( User user:[Select id, FirstName, LastName from user where Id IN  :ownerId])
    {   
        for(Case cas : [Select Id, OwnerId, First_Name__c, CaseNumber, AccountId From Case where AccountId IN:caseid])
        {        
            Case c = new Case();
            c.ID = cas.ID;
            c.OwnerId = user.Id;
            casesToUpdate.add(c);
        }  
    }

    update casesToUpdate;
}
